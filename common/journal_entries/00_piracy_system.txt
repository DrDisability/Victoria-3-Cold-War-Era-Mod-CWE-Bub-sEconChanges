je_piracy_system = {
	icon = "gfx/interface/icons/invention_icons/screw_frigate.dds"

	group = je_group_crises
	
	is_shown_when_inactive = {
		
		#Has port
		has_port = yes
		
		#Has piracy areas
		is_piracy_area = yes
		
	}
	
	possible = {
		
		#Too small a navy to fight piracy
		OR = { 
			navy_size < 50
			in_default = yes
		}
		
	}

	immediate = {
		
		#Apply piracy modifier
		every_scope_state = {
			limit = { 
				is_piracy_zone = yes 
				NOT = { has_modifier = piracy_ongoing }
			}
			add_modifier = {
				name = piracy_ongoing
			}
		}
		
	}
	
	on_yearly_pulse = {
		
		effect = {
		
			#Apply piracy modifier
			every_scope_state = {
				limit = { 
					is_piracy_zone = yes 
					NOT = { has_modifier = piracy_ongoing }
				}
				add_modifier = {
					name = piracy_ongoing
				}
			}
			
		}
		
	}

	complete = {
		
		OR = { 
			NOT = { is_piracy_area = yes }
			AND = {
				navy_size >= 50
				in_default = no
			}
		}
		
	}
	
	on_complete = {
		
		#Remove piracy modifier
		every_scope_state = {
			limit = { 
				has_modifier = piracy_ongoing
			}
			remove_modifier = piracy_ongoing
		}
		
	}
	
	invalid = {
		has_port = no
	}

	on_invalid = {
		
		#Remove piracy modifier
		every_scope_state = {
			limit = {
				has_modifier = piracy_ongoing
			}
			remove_modifier = piracy_ongoing
		}
		
	}

	weight = 100

	should_be_pinned_by_default = no

	transferable = no

	progressbar = no
	
}
