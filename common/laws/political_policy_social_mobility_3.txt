# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law


law_social_hierarchy_3_enforced = {
	group = lawgroup_social_mobility_3
	
	icon = "gfx/interface/icons/law_icons/social_hierarchy_enforced.dds"
	
	progressiveness = -75
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	can_enact = {
	
		#Constitutional Laws system (Requires simple majority to amend constitutional laws)
		government_has_legislative_super_majority = yes
		
		#No restriction by legal union
		THIS_is_not_restricted_by_confederal_law = yes
		
		#Cabinet agrees with this law
		custom_tooltip = {
			text = no_cabinet_objection.tt
			NOT = {
				any_interest_group = {
					OR = { 
						has_modifier = cabinet_minister_ig
						leader ?= { has_modifier = cabinet_minister }
					}
					law_stance = {
						law = law_type:law_social_hierarchy_3_enforced
						value < neutral
					}
				}
			}
		}
		
		#Ruler agrees with this
		ruler ?= {
			interest_group = {
				law_stance = {
					law = law_type:law_social_hierarchy_3_enforced
					value >= approve
				}
			}
		}
		
		#Not blocked by constitutional values
		custom_tooltip = {
			text = blocked_by_constitutional_value.tt
			NAND = {
				has_law = law_type:law_constitutional_supremacy
				OR = { 
					has_variable = liberty_nv_constitutionalised
					has_variable = equality_nv_constitutionalised
				}
			}
		}
		
	}

	modifier = {
		state_bureaucracy_population_base_cost_factor_mult = -0.20
		
		#Nationalist
		country_vanguard_class_qualification_growth_same_class_mult = -0.50
		country_business_class_qualification_growth_same_class_mult = -0.80
		country_petite_bourgeoisie_class_qualification_growth_same_class_mult = -0.80
		country_worker_class_qualification_growth_same_class_mult = -0.80
		country_culturally_discriminated_class_qualification_growth_same_class_mult = -0.80

		country_vanguard_class_qualification_growth_other_class_mult = -0.80
		country_business_class_qualification_growth_other_class_mult = -1
		country_petite_bourgeoisie_class_qualification_growth_other_class_mult = -1
		country_worker_class_qualification_growth_other_class_mult = -1
		country_culturally_discriminated_class_qualification_growth_other_class_mult = -1
		
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
