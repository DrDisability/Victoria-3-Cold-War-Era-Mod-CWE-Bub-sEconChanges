			value = global_var:superstate_customs_union_core_interest_rate
		}
		remove_variable = superstate_customs_union_core_interest_rate
		remove_variable = superstate_customs_union_core_interest_rate_members
	}
	
	#Determine level of inflation
	remove_variable = very_high_core_inflation
	remove_variable = high_core_inflation
	remove_variable = moderate_core_inflation
	remove_variable = low_core_inflation
	remove_variable = very_low_core_inflation
	if = {
		limit = { 
			root.var:core_interest_rate > 10
		}
		set_variable = very_high_core_inflation
	}
	if = {
		limit = { 
			root.var:core_interest_rate <= 10
			root.var:core_interest_rate > 5
		}
		set_variable = high_core_inflation
	}
	if = {
		limit = { 
			root.var:core_interest_rate <= 5
			root.var:core_interest_rate >= -5
		}
		set_variable = moderate_core_inflation
	}
	if = {
		limit = { 
			root.var:core_interest_rate < -5
			root.var:core_interest_rate > -10
		}
		set_variable = low_core_inflation
	}
	if = {
		limit = {
			root.var:core_interest_rate <= -10
		}
		set_variable = very_low_core_inflation
	}

}
